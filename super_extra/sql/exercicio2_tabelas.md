CREATE TABLE cidades(
    id NUMERIC PRIMARY KEY,
    nome VARCHAR(50),
    uf VARCHAR(2)
);

CREATE TABLE enderecos(
    id NUMERIC PRIMARY KEY,
    cep VARCHAR(10),
    rua VARCHAR(100),
    numero NUMERIC,
    complemento VARCHAR(100),
    bairro VARCHAR(100),
    cida_id NUMERIC,
    CONSTRAINT fk_enderecos_cidades
    FOREIGN KEY(cida_id)
    REFERENCES cidades(id)
);

CREATE TABLE clientes(
    id NUMERIC PRIMARY KEY,
    nome VARCHAR(50),
    email VARCHAR(100),
    cpf VARCHAR(11),
    telefone VARCHAR(11),
    data_cadastro TIMESTAMP,
    ende_id NUMERIC,
    CONSTRAINT fk_clientes_enderecos
    FOREIGN KEY (ende_id)
    REFERENCES enderecos(id)
);


--NOVO
DROP TABLE IF EXISTS clientes CASCADE;
DROP TABLE IF EXISTS enderecos CASCADE;
DROP TABLE IF EXISTS cidades CASCADE;
DROP TABLE IF EXISTS produtos CASCADE;

--sem FK
--validação de cidade ta no Java
CREATE TABLE cidades (
id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nome VARCHAR(50) NOT NULL,
uf CHAR(2) NOT NULL
);

CREATE TABLE enderecos (
id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
cep VARCHAR(8) NOT NULL,
rua VARCHAR(100) NOT NULL,
numero INTEGER NOT NULL,
complemento VARCHAR(100),
bairro VARCHAR(100) NOT NULL,
cida_id INTEGER
);

--sem FK pra permitir deletar endereço.
--validação de cidade existente no Java
CREATE TABLE clientes (
id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nome VARCHAR(50) NOT NULL,
email VARCHAR(100) NOT NULL,
cpf VARCHAR(11) NOT NULL UNIQUE,
telefone VARCHAR(11) NOT NULL,
data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
ende_id INTEGER
);

CREATE TABLE produtos (
id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nome VARCHAR(100) NOT NULL,
preco NUMERIC(10, 2) NOT NULL CHECK (preco >= 0),
categoria VARCHAR(50) NOT NULL,
estoque INTEGER NOT NULL CHECK (estoque >= 0)
);
